```python
def execute_with_rolling_plan(goal, max_replans=5):
    current_state = get_initial_state()
    replan_count = 0
    
    while not is_done(current_state):
        if replan_count >= max_replans:
            report_failure("超过最大重规划次数")
            return
        
        # 滚动规划（步长 3）
        plan = llm.plan(current_state, goal, max_steps=3)
        
        for step in plan:
            result = execute(step)
            expected = step.expected_output
            
            # 偏差检测
            if deviation(result, expected) > 0.3:
                log_deviation(step, result, expected)
                break  # 重新规划
            
            current_state = update(result)
        
        replan_count += 1
```
